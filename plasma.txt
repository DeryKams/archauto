# Обновление системы
sudo pacman -Syu

# Установка базового Xorg
sudo pacman -S xorg-server xorg-apps xorg-xinit

# ВЫБЕРИТЕ ВИДЕОДРАЙВЕР (пример для AMD):
sudo pacman -S xf86-video-amdgpu
# Для Intel: sudo pacman -S xf86-video-intel
# Для NVIDIA: sudo pacman -S nvidia nvidia-utils

# Установка Plasma и базовых приложений
sudo pacman -S plasma-meta plasma-workspace plasma-wayland-session ark dolphin dolphin-plugins kate konsole kscreen kinfocenter kdeconnect

# Системные компоненты
sudo pacman -S sddm sddm-kcm networkmanager pipewire pipewire-pulse wireplumber pipewire-alsa bluez bluez-utils cups system-config-printer tlp

# Включение служб
sudo systemctl enable sddm.service
sudo systemctl enable NetworkManager.service
sudo systemctl enable bluetooth.service
sudo systemctl enable cups.service
sudo systemctl enable tlp.service

# Дополнительные компоненты для удобства работы
sudo pacman -S firefox spectacle kcalc gwenview okular kate ark p7zip unrar unarchiver ttf-dejavu ttf-liberation noto-fonts neofetch htop
